/* Uevent sender for user-space notification */
static void send_saturation_uevent(int cpu, int saturation)
{
    char *envp[4];
    char cpu_buf[16];
    char sat_buf[16];
    char msg_buf[64];
    
    snprintf(cpu_buf, sizeof(cpu_buf), "CPU=%d", cpu);
    snprintf(sat_buf, sizeof(sat_buf), "SATURATION=%d", saturation);
    snprintf(msg_buf, sizeof(msg_buf), "ALERT=pre_rollback_saturation");
    
    envp[0] = cpu_buf;
    envp[1] = sat_buf;
    envp[2] = msg_buf;
    envp[3] = NULL;
